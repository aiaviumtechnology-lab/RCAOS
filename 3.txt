# Set up global logging framework with divided logs
def setup_logging(log_level=logging.DEBUG):
    """
    Configure the logging framework with console and multiple file handlers.
    - framework.log: For Workflow B (upgrades/errors).
    - workflow_a.log: For Workflow A.
    - system.log: For SystemAgent issues.
    - install.log: For installations (e.g., agent registrations).
    File logs rotate at 5MB, keeping 5 backups.
    log_level: The global log level (e.g., logging.INFO).
    """
    # Root logger
    root_logger = logging.getLogger()
    root_logger.setLevel(log_level)  # Set global level


    # Console handler for INFO and above
    console_handler = logging.StreamHandler()
    console_handler.setLevel(logging.INFO)
    console_formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
    console_handler.setFormatter(console_formatter)
    root_logger.addHandler(console_handler)


    # Helper to add rotating file handler
    def add_rotating_handler(logger_name, file_name, level=logging.DEBUG):
        logger = logging.getLogger(logger_name)
        logger.setLevel(level)
        handler = RotatingFileHandler(file_name, maxBytes=5*1024*1024, backupCount=5)
        handler.setLevel(level)
        formatter = logging.Formatter('%(asctime)s - %(name)s - %(levelname)s - %(message)s')
        handler.setFormatter(formatter)
        logger.addHandler(handler)
        logger.propagate = False  # Prevent propagation to root


    # Set up specific loggers
    add_rotating_handler('WorkflowA', 'workflow_a.log')
    add_rotating_handler('Framework', 'framework.log')  # For Workflow B
    add_rotating_handler('System', 'system.log')
    add_rotating_handler('Install', 'install.log')

class BaseAgent:
    """
    Base class for all agents. Handles communication via the A2A protocol through Orchestrator.
    Uses Llama3.2 for base reasoning (simulated).
    """
    def __init__(self, name, orchestrator=None, secret_key=None, llm_model="llama3.2", log_level=logging.DEBUG):
        self.name = name
        self.orchestrator = orchestrator
        self.secret_key = secret_key or b'secret_' + name.encode()  # Simulated per-agent secret key
        self.llm_model = llm_model
        self.logger = logging.getLogger(self.name)  # Per-agent logger
        self.logger.setLevel(log_level)  # Set per-agent log level
        # Generate RSA keys for encryption
        try:
            self.private_key = rsa.generate_private_key(public_exponent=65537, key_size=2048)
            self.public_key = self.private_key.public_key()
            self.logger.info("RSA keys generated successfully")
        except Exception as e:
            self.logger.error(f"Key generation error: {e}")
            raise EncryptionError("Failed to generate RSA keys")


    def set_log_level(self, level):
        """
        Set the log level for this agent's logger.
        level: str like 'DEBUG', 'INFO', etc., or logging constant.
        """
        if isinstance(level, str):
            level = getattr(logging, level.upper(), logging.DEBUG)
        self.logger.setLevel(level)
        self.logger.info(f"Log level set to {logging.getLevelName(level)}")


    def reason(self, prompt):
        """
        Use LLM for reasoning.
        """
        self.logger.debug(f"Reasoning with prompt: {prompt}")
        return call_llm(self.llm_model, prompt)

    def send(self, to_name, content):
        """
        Send a message to another agent using A2A protocol.
        Encrypts content with receiver's public key, creates structured message, signs it, and sends via Orchestrator.
        """
        self.logger.info(f"Sending message to {to_name}: {content}")
        if self.orchestrator is None:
            self.logger.error("No orchestrator set")
            return None
        try:
            if to_name not in self.orchestrator.public_keys:
                raise FrameworkError(f"Public key for {to_name} not found")
            receiver_public_key = self.orchestrator.public_keys[to_name]
            # Serialize and encrypt content
            serialized_content = json.dumps(content).encode()
            encrypted_content = receiver_public_key.encrypt(
                serialized_content,
                padding.OAEP(
                    mgf=padding.MGF1(algorithm=hashes.SHA256()),
                    algorithm=hashes.SHA256(),
                    label=None
                )
            )
            encrypted_b64 = base64.b64encode(encrypted_content).decode()
            # Create message
            message = {
                "sender": self.name,
                "receiver": to_name,
                "timestamp": time.time(),
                "encrypted_content": encrypted_b64
            }
            serialized = json.dumps(message, sort_keys=True).encode()
            signature = hmac.new(self.secret_key, serialized, hashlib.sha256).digest()
            message["signature"] = base64.b64encode(signature).decode()
            response = self.orchestrator.send_message(message)
            # Verify response before returning
            if self.verify_response(response):
                self.logger.debug(f"Response from {to_name}: {response}")
                print(f"{self.name} sent to {to_name}: {content} | Response: {response}")
                return response
            else:
                self.logger.error("Response verification failed")
                return "Invalid response"
        except EncryptionError as e:
            self.logger.error(f"Encryption error: {e}")
            return "Encryption failed"
        except FrameworkError as e:
            self.logger.error(f"Framework error: {e}")
            return str(e)
        except Exception as e:
            self.logger.error(f"Unexpected send error: {e}")
            return "Send failed"


    def verify_response(self, response):
        """
        Verify agent response (e.g., check for errors, validity).
        """
        # Simple check: Not None, not error string
        if response is None or "failed" in str(response).lower() or "error" in str(response).lower():
            return False
        return True


    def receive(self, message):
        """
        Receive a message, decrypt the content, and handle it.
        """
        self.logger.info(f"Received message from {message['sender']}")
        try:
            encrypted_b64 = message["encrypted_content"]
            encrypted = base64.b64decode(encrypted_b64)
            decrypted = self.private_key.decrypt(
                encrypted,
                padding.OAEP(
                    mgf=padding.MGF1(algorithm=hashes.SHA256()),
                    algorithm=hashes.SHA256(),
                    label=None
                )
            )
            content = json.loads(decrypted.decode())
            self.logger.debug(f"Decrypted content: {content}")
            return self.handle_content(message["sender"], content)
        except ValueError as e:
            self.logger.error(f"Base64 decode error: {e}")
            return "Invalid encrypted content"
        except Exception as e:
            self.logger.error(f"Decryption failed: {e}")
            return "Decryption error"


    def handle_content(self, sender, content):
        """
        Handle the decrypted content. Subclasses can override.
        """
        self.logger.info(f"Handling content from {sender}: {content}")
        print(f"{self.name} received from {sender}: {content}")
        return "Acknowledged"

    def send_message(self, message):
        """
        Route A2A message after verifying signature.
        """
        self.logger.debug(f"Routing message: {message}")
        try:
            to_name = message["receiver"]
            if to_name not in self.agents:
                raise FrameworkError(f"Agent {to_name} not found")
            # Verify signature
            sender_name = message.get("sender")
            sender_agent = self.agents.get(sender_name)
            if not sender_agent:
                raise SignatureError(f"Invalid sender: {sender_name}")
            
            # Recompute signature
            temp_message = message.copy()
            signature_b64 = temp_message.pop("signature")
            serialized = json.dumps(temp_message, sort_keys=True).encode()
            expected_signature = hmac.new(sender_agent.secret_key, serialized, hashlib.sha256).digest()
            received_signature = base64.b64decode(signature_b64)
            
            if not hmac.compare_digest(expected_signature, received_signature):
                raise SignatureError(f"Invalid signature from {sender_name} to {to_name}")
            
            # Signature valid, route to receiver
            return self.agents[to_name].receive(message)
        except SignatureError as e:
            self.logger.error(str(e))
            return str(e)
        except FrameworkError as e:
            self.logger.error(str(e))
            return str(e)
        except ValueError as e:
            self.logger.error(f"Base64 decode error in signature: {e}")
            return "Invalid signature format"
        except Exception as e:
            self.logger.error(f"Unexpected routing error: {e}")
            return "Routing failed"


    def handle_task(self, task):
        """
        Handle a user or system task: Route to ResearchAgent for tool lookup, then CI/CD Workflow A.
        """
        self.logger.info(f"Handling task: {task}")
        try:
            if "Research" not in self.agents:
                raise FrameworkError("No Research Agent registered")
            research_response = self.send("Research", {"type": "research_tool", "task": task})
            # Trigger CI/CD Workflow A
            self.cicd_workflow_a(task, research_response)
            return research_response
        except FrameworkError as e:
            self.logger.error(f"Task handling error: {e}")
            return None

        except WorkflowError as e:
            self.workflow_a_logger.error(f"CI/CD A error: {e}")
        except Exception as e:
            self.workflow_a_logger.error(f"Unexpected in CI/CD A: {e}")

class ResearchAgent(BaseAgent):
    """
    Agent for researching tools needed for tasks.
    Uses ToT for exploring multiple research paths.
    Queries DatabaseAdministrator for tools; if not found, requests CodingAgent.
    """
    def handle_content(self, sender, content):
        self.logger.info(f"Handling research content: {content}")
        try:
            if isinstance(content, dict) and content.get("type") == "research_tool":
                task = content["task"]
                # Implement ToT: Generate multiple thoughts, evaluate
                thoughts = self.tree_of_thoughts(f"Research paths for {task}", num_thoughts=3)
                if not thoughts:
                    raise FrameworkError("No thoughts generated in ToT")
                best_thought = max(thoughts, key=lambda t: random.random())  # Simulate evaluation
                print(f"Best research path: {best_thought}")
                self.logger.debug(f"Best thought: {best_thought}")
                # Query DBA
                response = self.send("Database", {"type": "search_tool", "query": task})
                if response == "Tool not found":
                    # Request CodingAgent
                    self.send("Coding", {"type": "code_tool", "outline": task})
                    # Simulate notification after coding
                    time.sleep(1)  # Simulate delay
                    self.orchestrator.tool_database[task] = f"Tool for {task} created."
                    return "Tool researched and added."
                else:
                    return "Tool found."
            return "Acknowledged"
        except FrameworkError as e:
            self.logger.error(f"Research error: {e}")
            return "Research failed"


    def tree_of_thoughts(self, prompt, num_thoughts=3, depth=2):
        """
        Tree of Thoughts: Generate branches of thoughts, evaluate.
        """
        self.logger.debug(f"Starting ToT with prompt: {prompt}")
        try:
            thoughts = []
            for _ in range(num_thoughts):
                thought = self.reason(f"Generate thought for {prompt}")
                for _ in range(depth - 1):
                    thought += " -> " + self.reason(f"Expand {thought}")
                thoughts.append(thought)
            # Simulate evaluation
            self.logger.debug(f"ToT thoughts: {thoughts}")
            return thoughts
        except Exception as e:
            self.logger.error(f"ToT error: {e}")
            return []
